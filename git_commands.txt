cat ~/.ssh/id_rsa.pub - полчение ssh ключа

git - репозиторий

git commit -m "message for commit" - создает версии кода с сообщением 

git remote add <name_of_remote> <ssh_key> - свзывание локального репозитория с удаленным

git push <name_of remote> <branch_name> - отправка версии на гитхаб
